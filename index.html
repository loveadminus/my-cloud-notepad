// 彻底注销账号并删除云端数据 async function deleteAccount() { if(!cfUser) return; // 严厉的防误触警告机制 const confirmText = prompt(`🚨 危险操作警告 🚨\n此操作将不可逆转地彻底销毁您的账号【${cfUser}】以及云端所有的笔记和分类数据！\n如果您确定要销毁，请在下方输入框中准确输入【确认注销】：`); if (confirmText !== "确认注销") { if(confirmText !== null) alert("输入不匹配，已安全取消注销操作。"); return; } try { const res = await fetch('/api/delete-account', { method: 'POST', headers: getAuthHeaders() }); const data = await res.json(); if(res.ok && data.success) { alert("销毁成功：您的账号及所有云端数据已从服务器彻底清除！"); localStorage.removeItem('cfUser'); localStorage.removeItem('cfPass'); location.reload(); // 自动刷新回登入界面 } else { alert("注销失败：" + (data.message || "未知错误")); } } catch(e) { alert("网络请求失败：" + e.message); } }
